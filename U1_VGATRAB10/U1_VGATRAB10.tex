\textbf{(a) $D + E$}
\begin{align*}
    D + E &=
    \underbrace{
    \begin{bmatrix}
        1 & 5 & 2\\
        -1  & 0 & 1\\
        3 & 2 & 4
    \end{bmatrix}}_{D} +
    \underbrace{
    \begin{bmatrix}
        6 & 1 & 3\\
        -1 & 1  & 2\\
        4 & 1  & 3
    \end{bmatrix}}_{E} 
    =
    \begin{bmatrix}
            1+6 & 5+2 & 2+3\\
            -1+(-1)  & 0+1 &1+2\\
            3+4 & 2+1 & 4+3
    \end{bmatrix}
    =
    \underbrace{\begin{bmatrix}
        7 & 7 & 5\\
        -2 & 1 & 3\\
        7 & 3 & 7
    \end{bmatrix}}_{D + E}
\end{align*}
    
\textbf{(b) $D - E$}
\begin{align*}
    D + E &=
    \underbrace{
    \begin{bmatrix}
        1 & 5 & 2\\
        -1  & 0 & 1\\
        3 & 2 & 4
    \end{bmatrix}}_{D} -
    \underbrace{
    \begin{bmatrix}
        6 & 1 & 3\\
        -1 & 1  & 2\\
        4 & 1  & 3
    \end{bmatrix}}_{E} 
    =
    \begin{bmatrix}
            1-6 & 5-2 & 2-3\\
            -1-(-1)  & 0-1 &1-2\\
            3-4 & 2-1 & 4-3
    \end{bmatrix}
    =
    \underbrace{\begin{bmatrix}
        -5 & 3 & -1\\
        0 & -1 & -1\\
        -1 & 1 & 1
    \end{bmatrix}}_{D - E}
\end{align*}
